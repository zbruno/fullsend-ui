## Install

```bash
npm install --save sendit
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'sendit'
import 'sendit/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
